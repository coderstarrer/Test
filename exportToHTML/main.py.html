import math

print("Welcome to the Hypotenuse Calculator")
print("------------------------------------")

# Hypo in this statement means Hypotenuse furthermore The II is the latin number(e.g i = 1 ii=2 iii=3 iv=4 and so on)

side_hypo = float(input("Enter Side A:"))
side_hypoII = float(input("Enter Side B:"))

# Restenuse is a combination of the word result and hypotenuse (res+tenuse)

restenuse = math.sqrt(pow(side_hypo, 2) + pow(side_hypoII, 2))
print("The result is:" + str(restenuse))
